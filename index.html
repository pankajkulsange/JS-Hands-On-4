const Find_Prod = (array, N) => 
{
     let prod = 1;
     for(let i = 0; i < N; i++){

{

const findEvenOdd = (N, Arr) => 
{
    let B = [0, 0];
    for(let i = 0; i < N; i++){
      if(Arr[i] % 2 === 0){
        B[0] += Arr[i];
      }
      else {
        B[1] += Arr[i];
      }
    }
    return B;
};

Que11: Birthday Game

Ans:

function Birthday_Game(arr,D ,M) {
let count = 0;
for(i = 0; i < arr.length-M+1; i++){
  let sum = 0;
  for(j = i; j < M+i; j++){
    sum += arr[j];
  }
  if(sum === D){
    count++;
  }
}
return count;
}
